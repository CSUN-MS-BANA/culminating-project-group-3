# year-over-year changes
countries = data['country_name'].unique()
results = []

for country in countries:
    country_data = data[data['country_name'] == country].sort_values('year')

    if len(country_data) < 2:
        continue

    for i in range(1, len(country_data)):
        year_prev = country_data.iloc[i-1]['year']
        year_curr = country_data.iloc[i]['year']

        co2_change = country_data.iloc[i]['CO2Emissions_Agriculture'] - country_data.iloc[i-1]['CO2Emissions_Agriculture']
        life_exp_change = country_data.iloc[i]['LifeExpectancy_Total'] - country_data.iloc[i-1]['LifeExpectancy_Total']

        results.append({
            'country_name': country,
            'Region': country_data.iloc[i]['region'],
            'year_prev': year_prev,
            'year_curr': year_curr,
            'co2_change': co2_change,
            'life_exp_change': life_exp_change
        })

changes_df = pd.DataFrame(results)

# Visualize the relationship between changes in CO2 Emissions-Agriculture and changes in life expectancy
plt.figure(figsize=(12, 8))
sns.scatterplot(data=changes_df, x='co2_change', y='life_exp_change', hue='Region', alpha=0.7)
plt.title('Relationship Between Changes in CO2 Emissions-Agriculture and Changes in Life Expectancy', fontsize=14)
plt.xlabel('Change in CO2 Emissions-Agriculture')
plt.ylabel('Change in Life Expectancy (years)')
plt.grid(True)
plt.legend(bbox_to_anchor=(1.05, 1), loc=2)
plt.tight_layout()
plt.show()
